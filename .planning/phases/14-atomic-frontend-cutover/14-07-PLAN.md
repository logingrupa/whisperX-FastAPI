    ```ts
     * API client for task management — refactored Phase 14 to use apiClient.

     */
    import type { ApiResult } from '@/types/api';
    import type { TaskResult } from '@/types/transcript';
    import { apiClient, ApiClientError, RateLimitError, AuthRequiredError } from '@/lib/apiClient';

    /**
     * Fetch task result including transcript segments.
     * Retrieves full task details from GET /task/{identifier}.
     */
    export async function fetchTaskResult(taskId: string): Promise<ApiResult<TaskResult>> {
      try {
        const data = await apiClient.get<TaskResult>(`/task/${taskId}`);
        return { success: true, data };
      } catch (err) {
        if (err instanceof AuthRequiredError) {
          // apiClient already redirected the page — propagate so the catch flows
          throw err;
        }
        if (err instanceof RateLimitError) {
          return {
            success: false,
            error: { status: 429, detail: `Rate limited; retry in ${err.retryAfterSeconds}s` },
          };
        }
        if (err instanceof ApiClientError) {
          return { success: false, error: { status: err.status, detail: err.message } };
        }
        return {
          success: false,
          error: { status: 0, detail: err instanceof Error ? err.message : 'Network error' },
        };
      }
    }
    ```

    ```ts
        }
        return {
          success: false,
          error: { status: 0, detail: err instanceof Error ? err.message : 'Network error' },
        };
      }
    }
    ```

    ```ts
    /**
     * WebSocket ticket helper (MID-06 enforcement on the client).
     *
     * Backend rejects WS connections without a single-use 60-second ticket.
     * This module provides:
     *   - requestWsTicket(taskId): get ticket via apiClient.post('/api/ws/ticket')
     *   - buildTaskSocketUrl(taskId): convenience that returns the ws URL with ?ticket
     *
     * Tickets are single-use + 60s TTL — callers MUST re-request on reconnect.
     */

    import { apiClient } from '@/lib/apiClient';

    interface WsTicketResponse {
      ticket: string;
      expires_at: string;
    }

    export async function requestWsTicket(taskId: string): Promise<string> {
      const response = await apiClient.post<WsTicketResponse>('/api/ws/ticket', {
        task_id: taskId,
      });
      return response.ticket;
    }

    export async function buildTaskSocketUrl(taskId: string): Promise<string> {
      const ticket = await requestWsTicket(taskId);
      return `/ws/tasks/${taskId}?ticket=${encodeURIComponent(ticket)}`;
    }
    ```
