    ```python
    class SQLAlchemyTaskRepository:
        def __init__(self, session: Session):
            self.session = session
            self._user_scope: int | None = None

        def set_user_scope(self, user_id: int | None) -> None:
            self._user_scope = user_id

        def _scoped_query(self):
            """Return a base query filtered by user_scope if set."""
            query = self.session.query(ORMTask)
            if self._user_scope is not None:
                query = query.filter(ORMTask.user_id == self._user_scope)
            return query

        def add(self, task: DomainTask) -> str:
            try:
                if not task.uuid:
                    task.uuid = str(uuid4())
                # Inject scope user_id if writer hasn't set it explicitly
                if task.user_id == 0 and self._user_scope is not None:
                    task.user_id = self._user_scope
                orm_task = to_orm(task)
                self.session.add(orm_task)
                self.session.commit()
                self.session.refresh(orm_task)
                logger.info(f"Task added successfully with UUID: {orm_task.uuid} user_id={orm_task.user_id}")
                return str(orm_task.uuid)
            except SQLAlchemyError as e:
                self.session.rollback()
                logger.error(f"Failed to add task: {str(e)}")
                raise DatabaseOperationError(operation="add", reason=str(e), original_error=e, identifier=task.uuid)

        def get_by_id(self, identifier: str) -> DomainTask | None:
            try:
                orm_task = self._scoped_query().filter(ORMTask.uuid == identifier).first()
                if orm_task is None:
                    return None
                return to_domain(orm_task)
            except SQLAlchemyError as e:
                logger.error(f"Failed to get task by ID {identifier}: {str(e)}")
                return None

        def get_all(self) -> list[DomainTask]:
            try:
                orm_tasks = self._scoped_query().all()
                return [to_domain(t) for t in orm_tasks]
            except SQLAlchemyError as e:
                logger.error(f"Failed to get all tasks: {str(e)}")
                return []

        def update(self, identifier: str, update_data: dict[str, Any]) -> None:
            orm_task = self._scoped_query().filter(ORMTask.uuid == identifier).first()
            if not orm_task:
                logger.error(f"Task not found for update with UUID: {identifier}")
                raise ValueError(f"Task not found with UUID: {identifier}")
            try:
                for key, value in update_data.items():
                    if hasattr(orm_task, key):
                        setattr(orm_task, key, value)
                self.session.commit()
            except SQLAlchemyError as e:
                self.session.rollback()
                raise DatabaseOperationError(operation="update", reason=str(e), original_error=e, identifier=identifier)

        def delete(self, identifier: str) -> bool:
            try:
                orm_task = self._scoped_query().filter(ORMTask.uuid == identifier).first()
                if orm_task is None:
                    return False
                self.session.delete(orm_task)
                self.session.commit()
                return True
            except SQLAlchemyError as e:
                self.session.rollback()
                logger.error(f"Failed to delete task {identifier}: {str(e)}")
                return False
    ```

    2a. Add `get_scoped_task_repository` to `app/api/dependencies.py`:

    ```python
    def get_scoped_task_repository(request: Request) -> Generator[ITaskRepository, None, None]:
        """Yield a task repository scoped to the authenticated user's id.

        DualAuthMiddleware sets request.state.user; we resolve user.id and call
        repo.set_user_scope(user.id) before yielding. Every read/write the route
        performs is automatically WHERE user_id = caller.

        Cross-user requests naturally return None/404 because the scope filter
        excludes them at the SQL layer.
        """
        if _container is None:
            raise RuntimeError(CONTAINER_NOT_INITIALIZED_ERROR)
        user = getattr(request.state, "user", None)
        if user is None:
            from fastapi import HTTPException, status
            raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Authentication required")
        repository = _container.task_repository()
        repository.set_user_scope(int(user.id))
        try:
            yield repository
        finally:
            repository.set_user_scope(None)  # tidy in case the Factory got pooled (defensive)
    ```
