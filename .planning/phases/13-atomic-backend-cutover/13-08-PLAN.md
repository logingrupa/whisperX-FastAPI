    ```python
        def release(
            self,
            bucket_key: str,
            *,
            tokens: int = 1,
            capacity: int = 1,
        ) -> None:
            """Refund `tokens` to bucket (capped at `capacity`).

            Used to release a held slot — e.g. concurrency semaphore slot
            held while a transcription runs. Caller passes the same `capacity`
            it used in the matching check_and_consume() call.

            No-op if bucket_key absent (defensive — release without prior
            consume should never crash).
            """
            existing = self.repository.get_by_key(bucket_key)
            if existing is None:
                logger.debug("RateLimit release no-op (no bucket) key=%s", bucket_key)
                return
            new_tokens = min(capacity, existing.tokens + tokens)
            self.repository.upsert_atomic(
                bucket_key,
                {"tokens": new_tokens, "last_refill": existing.last_refill},
            )
            logger.debug(
                "RateLimit released bucket=%s tokens=%d/%d",
                bucket_key, new_tokens, capacity,
            )
    ```

    1c. Create `app/services/free_tier_gate.py`:

    ```python
    """FreeTierGate — enforces RATE-03..RATE-10 in fail-fast order (CONTEXT §137-145).

    Per-tier policy table:
      free:  5/hr, 5min file, 30min/day, tiny+small, no diarize, 1 concurrent
      pro:   100/hr, 60min file, 600min/day, all models, diarize OK, 3 concurrent
      trial: identical to free; expires at trial_started_at + 7 days → 402

    Concurrency slot lifecycle:
      - check() consumes 1 token from `user:{id}:concurrent` (capacity = max_concurrent)
      - process_audio_common's completion hook (success OR failure) calls
        FreeTierGate.release_concurrency(user) to refund the slot.
    """
    from __future__ import annotations
    import logging
    from dataclasses import dataclass
    from datetime import datetime, timedelta, timezone

    from app.core.exceptions import (
        ConcurrencyLimitError,
        FreeTierViolationError,
        RateLimitExceededError,
        TrialExpiredError,
    )
    from app.domain.entities.user import User
    from app.services.auth.rate_limit_service import RateLimitService

    logger = logging.getLogger(__name__)

    TRIAL_DAYS = 7


    @dataclass(frozen=True)
    class TierPolicy:
        max_per_hour: int
        max_file_seconds: int
        max_daily_seconds: int
        allowed_models: frozenset[str]
        diarization_allowed: bool
        max_concurrent: int


    FREE_POLICY = TierPolicy(
        max_per_hour=5,
        max_file_seconds=5 * 60,
        max_daily_seconds=30 * 60,
        allowed_models=frozenset({"tiny", "small"}),
        diarization_allowed=False,
        max_concurrent=1,
    )

    PRO_POLICY = TierPolicy(
        max_per_hour=100,
        max_file_seconds=60 * 60,
        max_daily_seconds=600 * 60,
        allowed_models=frozenset({"tiny", "base", "small", "medium", "large", "large-v2", "large-v3"}),
        diarization_allowed=True,
        max_concurrent=3,
    )


    def concurrency_bucket_key(user_id: int) -> str:
        """Single source of truth for the concurrency slot key."""
        return f"user:{user_id}:concurrent"


    class FreeTierGate:
        """Enforce free / pro / trial tier policies (CONTEXT §137-145)."""

        def __init__(self, rate_limit_service: RateLimitService) -> None:
            self.rate_limit_service = rate_limit_service

        def _policy_for(self, user: User) -> TierPolicy:
            if user.plan_tier == "pro":
                return PRO_POLICY
            return FREE_POLICY

        def _check_trial_expiry(self, user: User) -> None:
            if user.plan_tier != "trial":
                return
            if user.trial_started_at is None:
                return
            now = datetime.now(timezone.utc)
            if user.trial_started_at + timedelta(days=TRIAL_DAYS) < now:
                raise TrialExpiredError()

        def check(
            self,
            *,
            user: User,
            file_seconds: float,
            model: str,
            diarize: bool,
        ) -> None:
            """Run the 6 fail-fast gates. Raise on first failure.

            Order (CONTEXT §138):
              1. hourly transcribe rate (429)
              2. trial expiry (402)
              3. file duration (422)
              4. daily audio min cap (429)
              5. allowed model (403)
              6. concurrency slot acquired (429)
            """
            policy = self._policy_for(user)
            user_id = int(user.id)
            self._check_trial_expiry(user)
            self._check_hourly_rate(user_id, policy)
            self._check_file_duration(file_seconds, policy)
            self._check_model(model, policy)
            self._check_diarization(diarize, policy)
            self._check_daily_minutes(user_id, file_seconds, policy)
            self._check_concurrency(user_id, policy)

        def check_diarize_route(self, user: User) -> None:
            """Pro-only diarize route guard (no transcribe rate hit)."""
            self._check_trial_expiry(user)
            if not self._policy_for(user).diarization_allowed:
                raise FreeTierViolationError("Diarization not available on your plan")

        def release_concurrency(self, user: User) -> None:
            """Release 1 concurrency slot for `user`. Must be called from the
            transcription completion path (success AND failure) — failure to
            release locks the user out of further transcribes."""
            policy = self._policy_for(user)
            user_id = int(user.id)
            self.rate_limit_service.release(
                concurrency_bucket_key(user_id),
                tokens=1,
                capacity=policy.max_concurrent,
            )

        def _check_hourly_rate(self, user_id: int, policy: TierPolicy) -> None:
            allowed = self.rate_limit_service.check_and_consume(
                f"user:{user_id}:tx:hour",
                tokens_needed=1,
                rate=policy.max_per_hour / 3600.0,
                capacity=policy.max_per_hour,
            )
            if not allowed:
                raise RateLimitExceededError(bucket_key=f"user:{user_id}:tx:hour", retry_after_seconds=60)

        def _check_file_duration(self, file_seconds: float, policy: TierPolicy) -> None:
            if file_seconds > policy.max_file_seconds:
                raise FreeTierViolationError(
                    f"File duration {int(file_seconds)}s exceeds tier limit {policy.max_file_seconds}s"
                )

        def _check_model(self, model: str, policy: TierPolicy) -> None:
            if model not in policy.allowed_models:
                raise FreeTierViolationError(f"Model '{model}' not available on your plan")

        def _check_diarization(self, diarize: bool, policy: TierPolicy) -> None:
            if diarize and not policy.diarization_allowed:
                raise FreeTierViolationError("Diarization not available on your plan")

        def _check_daily_minutes(self, user_id: int, file_seconds: float, policy: TierPolicy) -> None:
            tokens_needed = max(1, int(file_seconds / 60))
            allowed = self.rate_limit_service.check_and_consume(
                f"user:{user_id}:audio_min:day",
                tokens_needed=tokens_needed,
                rate=policy.max_daily_seconds / 86400.0,
                capacity=policy.max_daily_seconds // 60,
            )
            if not allowed:
                raise RateLimitExceededError(bucket_key=f"user:{user_id}:audio_min:day", retry_after_seconds=3600)

        def _check_concurrency(self, user_id: int, policy: TierPolicy) -> None:
            # Slot is held until process_audio_common completion hook calls
            # release_concurrency(user). rate=0 — no automatic refill; release
            # is the only path back to a full bucket.
            allowed = self.rate_limit_service.check_and_consume(
                concurrency_bucket_key(user_id),
                tokens_needed=1,
                rate=0.0,
                capacity=policy.max_concurrent,
            )
            if not allowed:
                raise ConcurrencyLimitError()
    ```

    ```python
    """Per-completed-transcription usage_events writer (RATE-11, BILL-04)."""
    from __future__ import annotations
    import logging
    from datetime import datetime, timezone

    from sqlalchemy import text
    from sqlalchemy.exc import IntegrityError
    from sqlalchemy.orm import Session

    logger = logging.getLogger(__name__)


    class UsageEventWriter:
        """Idempotent insert into usage_events (idempotency_key=task.uuid)."""

        def __init__(self, session: Session) -> None:
            self.session = session

        def record(
            self,
            *,
            user_id: int,
            task_uuid: str,
            gpu_seconds: float,
            file_seconds: float,
            model: str,
        ) -> None:
            """Insert one usage_events row. Idempotent — duplicate task_uuid is no-op."""
            try:
                self.session.execute(
                    text(
                        "INSERT INTO usage_events "
                        "(user_id, task_id, gpu_seconds, file_seconds, model, idempotency_key, created_at) "
                        "VALUES (:user_id, :task_id, :gpu_seconds, :file_seconds, :model, :idempotency_key, :created_at)"
                    ),
                    {
                        "user_id": user_id,
                        "task_id": task_uuid,
                        "gpu_seconds": gpu_seconds,
                        "file_seconds": file_seconds,
                        "model": model,
                        "idempotency_key": task_uuid,
                        "created_at": datetime.now(timezone.utc),
                    },
                )
                self.session.commit()
                logger.info("usage_events recorded user_id=%s task_id=%s", user_id, task_uuid)
            except IntegrityError:
                self.session.rollback()
                logger.debug("usage_events duplicate skipped task_id=%s", task_uuid)
    ```
