    """Phase 13 end-to-end smoke test — single sequenced flow + feature flag check.

    Spawns a uvicorn subprocess with V2 ON + tmp DB; exercises:
      - register → cookie set
      - login → cookie set
      - create key (show-once)
      - use key as Bearer to call /api/keys
      - delete key (soft-delete)
      - logout (cookies cleared)
      - cross-user 404 isolation
      - rate-limit ANTI-01 (4 register in row)
      - free-tier RATE-03 (6th transcribe in hour)
      - disposable-email rejection
      - CORS preflight allowlist + reject
      - billing stubs 501
      - V2_ENABLED=false: /auth/register returns 404
    """
    from __future__ import annotations

    import os
    import secrets
    import shutil
    import socket
    import subprocess
    import sys
    import time
    from pathlib import Path

    import httpx
    import pytest


    def _free_port() -> int:
        with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as sock:
            sock.bind(("127.0.0.1", 0))
            return int(sock.getsockname()[1])


    def _start_server(env_overrides: dict[str, str], db_path: Path, port: int) -> subprocess.Popen:
        env = os.environ.copy()
        env["DB_URL"] = f"sqlite:///{db_path}"
        env["AUTH__JWT_SECRET"] = secrets.token_urlsafe(32)
        env["AUTH__CSRF_SECRET"] = secrets.token_urlsafe(32)
        env["AUTH__FRONTEND_URL"] = "http://localhost:5173"
        env["AUTH__COOKIE_SECURE"] = "false"
        env["ENVIRONMENT"] = "development"
        env.update(env_overrides)
        # Run alembic upgrade in same env
        subprocess.run(
            [sys.executable, "-m", "alembic", "upgrade", "head"],
            env=env, check=True, capture_output=True, text=True,
        )
        proc = subprocess.Popen(
            [sys.executable, "-m", "uvicorn", "app.main:app", "--host", "127.0.0.1", "--port", str(port)],
            env=env,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
        )
        # Wait for boot — poll /health
        for _ in range(30):
            time.sleep(0.5)
            try:
                r = httpx.get(f"http://127.0.0.1:{port}/health", timeout=1.0)
                if r.status_code == 200:
                    return proc
            except httpx.RequestError:
                continue
        proc.kill()
        out, err = proc.communicate()
        raise RuntimeError(f"Server did not boot in 15s. stderr:\n{err.decode()}")


    @pytest.fixture

    @pytest.mark.integration
    def test_full_register_login_create_key_use_key_logout_flow(server_v2_on: str) -> None:
        """The end-to-end happy path: register → create key → bearer-auth → revoke → logout."""
        with httpx.Client(base_url=server_v2_on) as client:
            # 1. Register
            r = client.post("/auth/register", json={"email": "alice@example.com", "password": "supersecret123"})
            assert r.status_code == 201, r.text
            assert "session" in r.cookies
            assert "csrf_token" in r.cookies
            cookies = dict(r.cookies)
            csrf = cookies["csrf_token"]

            # 2. Create API key (cookie auth + CSRF)
            r = client.post(
                "/api/keys",
                json={"name": "my key"},
                cookies=cookies,
                headers={"X-CSRF-Token": csrf},
            )
            assert r.status_code == 201, r.text
            payload = r.json()
            assert payload["key"].startswith("whsk_")
            assert len(payload["key"]) == 36
            api_key = payload["key"]
            key_id = payload["id"]

            # 3. List keys via cookie auth
            r = client.get("/api/keys", cookies=cookies)
            assert r.status_code == 200
            items = r.json()
            assert len(items) == 1
            assert "key" not in items[0]  # show-once: no plaintext in list

            # 4. Use bearer key to list keys (no cookie)
            r = client.get("/api/keys", headers={"Authorization": f"Bearer {api_key}"})
            assert r.status_code == 200
            assert len(r.json()) == 1

            # 5. Revoke key (soft-delete)
            r = client.delete(
                f"/api/keys/{key_id}",
                cookies=cookies,
                headers={"X-CSRF-Token": csrf},
            )
            assert r.status_code == 204

            # 6. List again — key shows as revoked
            r = client.get("/api/keys", cookies=cookies)
            assert r.status_code == 200
            assert r.json()[0]["status"] == "revoked"

            # 7. Logout — cookies cleared
            r = client.post("/auth/logout", cookies=cookies, headers={"X-CSRF-Token": csrf})
            assert r.status_code == 204
            # session cookie should be cleared in response
            assert "session" in r.cookies or "Max-Age=0" in (r.headers.get("set-cookie") or "")


    @pytest.mark.integration
    def test_cross_user_404_isolation(server_v2_on: str) -> None:
        with httpx.Client(base_url=server_v2_on) as client:
            # Register two users
            r1 = client.post("/auth/register", json={"email": "a@example.com", "password": "supersecret123"})
            assert r1.status_code == 201
            cookies_a = dict(r1.cookies)

            r2 = client.post("/auth/register", json={"email": "b@example.com", "password": "supersecret123"})
            assert r2.status_code == 201
            cookies_b = dict(r2.cookies)

            # User A creates a key
            ra = client.post("/api/keys", json={"name": "a-key"}, cookies=cookies_a, headers={"X-CSRF-Token": cookies_a["csrf_token"]})
            a_key_id = ra.json()["id"]

            # User B tries DELETE A's key → 404 opaque
            rb = client.delete(f"/api/keys/{a_key_id}", cookies=cookies_b, headers={"X-CSRF-Token": cookies_b["csrf_token"]})
            assert rb.status_code == 404


    @pytest.mark.integration
    def test_register_rate_limit_3_per_hour(server_v2_on: str) -> None:
        with httpx.Client(base_url=server_v2_on) as client:
            # 3 registers from same /24 — succeed
            for i in range(3):
                r = client.post("/auth/register", json={"email": f"user{i}@example.com", "password": "supersecret123"})
                assert r.status_code in (201, 422), f"unexpected {r.status_code}"
            # 4th — 429 with Retry-After
            r = client.post("/auth/register", json={"email": "user4@example.com", "password": "supersecret123"})
            assert r.status_code == 429
            assert "retry-after" in {k.lower() for k in r.headers.keys()}

    @pytest.mark.integration
    def test_cors_preflight_allowed_origin(server_v2_on: str) -> None:
        with httpx.Client(base_url=server_v2_on) as client:
            r = client.options(
                "/auth/register",
                headers={
                    "Origin": "http://localhost:5173",
                    "Access-Control-Request-Method": "POST",
                    "Access-Control-Request-Headers": "content-type",
                },
            )
            assert r.headers.get("access-control-allow-origin") == "http://localhost:5173"
            assert r.headers.get("access-control-allow-credentials") == "true"


    @pytest.mark.integration
    def test_cors_preflight_disallowed_origin(server_v2_on: str) -> None:
        with httpx.Client(base_url=server_v2_on) as client:
            r = client.options(
                "/auth/register",
                headers={
                    "Origin": "https://evil.example.com",
                    "Access-Control-Request-Method": "POST",
                },
            )
            # CORSMiddleware does NOT echo the origin for non-allowlisted requests
            assert r.headers.get("access-control-allow-origin") != "https://evil.example.com"


    @pytest.mark.integration
    def test_billing_stubs_return_501(server_v2_on: str) -> None:
        with httpx.Client(base_url=server_v2_on) as client:
            # Register first to get auth cookie for /billing/checkout (auth required)
            client.post("/auth/register", json={"email": "z@example.com", "password": "supersecret123"})
            cookies = client.cookies
            r = client.post("/billing/checkout", json={"plan": "pro"}, cookies=cookies, headers={"X-CSRF-Token": cookies["csrf_token"]})
            assert r.status_code == 501
            body = r.json()
            assert body.get("status") == "stub"
            assert "v1.3" in body.get("hint", "")

            # /billing/webhook with valid header schema
            r = client.post("/billing/webhook", headers={"Stripe-Signature": "t=1700000000,v1=abc123"})
            assert r.status_code == 501
            # /billing/webhook with malformed header
            r = client.post("/billing/webhook", headers={"Stripe-Signature": "garbage"})
            assert r.status_code == 400


    @pytest.mark.integration
    def test_v2_disabled_routes_not_registered(server_v2_off: str) -> None:
        with httpx.Client(base_url=server_v2_off) as client:
            # /auth/register is NOT registered when V2=off
            r = client.post("/auth/register", json={"email": "x@example.com", "password": "supersecret123"})
            assert r.status_code == 404
            # /health still works
            r = client.get("/health")
            assert r.status_code == 200


    @pytest.mark.integration
    def test_v2_enabled_routes_registered(server_v2_on: str) -> None:
        with httpx.Client(base_url=server_v2_on) as client:
            # /auth/register registered — empty body returns 422 not 404
            r = client.post("/auth/register", json={})
            assert r.status_code == 422


    @pytest.mark.integration
    def test_csrf_required_on_cookie_post(server_v2_on: str) -> None:
        with httpx.Client(base_url=server_v2_on) as client:
            r = client.post("/auth/register", json={"email": "csrf@example.com", "password": "supersecret123"})
            assert r.status_code == 201
            cookies = dict(r.cookies)
            # POST /api/keys WITHOUT X-CSRF-Token → 403
            r = client.post("/api/keys", json={"name": "no-csrf"}, cookies=cookies)
            assert r.status_code == 403


    @pytest.mark.integration
    def test_bearer_auth_skips_csrf(server_v2_on: str) -> None:
        with httpx.Client(base_url=server_v2_on) as client:
            # Register and create a key first via cookie + CSRF
            r = client.post("/auth/register", json={"email": "bearer@example.com", "password": "supersecret123"})
            cookies = dict(r.cookies)
            r = client.post("/api/keys", json={"name": "bk"}, cookies=cookies, headers={"X-CSRF-Token": cookies["csrf_token"]})
            api_key = r.json()["key"]
            # Now POST another key using Bearer (no CSRF) — should succeed
            r = client.post("/api/keys", json={"name": "bk2"}, headers={"Authorization": f"Bearer {api_key}"})
            assert r.status_code == 201
