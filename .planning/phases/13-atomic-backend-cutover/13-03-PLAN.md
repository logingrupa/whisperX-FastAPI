    ```python
    """Disposable-email domain blocklist (ANTI-04).

    Loads data/disposable-emails.txt into a frozenset at module import. O(1) lookup.
    Refresh: server restart (no hot reload — file is bundled).
    """
    from __future__ import annotations
    from pathlib import Path

    _BLOCKLIST_PATH = Path(__file__).resolve().parent.parent.parent / "data" / "disposable-emails.txt"


    def _load_blocklist() -> frozenset[str]:
        if not _BLOCKLIST_PATH.exists():
            return frozenset()
        return frozenset(
            line.strip().lower()
            for line in _BLOCKLIST_PATH.read_text(encoding="utf-8").splitlines()
            if line.strip()
        )


    DISPOSABLE_DOMAINS: frozenset[str] = _load_blocklist()


    def is_disposable(email: str) -> bool:
        """Return True if the email's domain is in the blocklist (case-insensitive)."""
        if "@" not in email:
            return False
        domain = email.rsplit("@", 1)[1].strip().lower()
        return domain in DISPOSABLE_DOMAINS
    ```

    SRP — single function. DRT — single load site. No nested-if.

    **1c. `app/core/rate_limiter.py`** — slowapi Limiter with custom key_func (CONTEXT §107):

    ```python
    """Slowapi Limiter with CF-Connecting-IP-aware key_func.

    Per RATE-01 + CONTEXT §107:
      - When AUTH__TRUST_CF_HEADER=true, prefer X-Forwarded-For/CF-Connecting-IP.
      - Group IPv4 by /24 and IPv6 by /64.
      - Backend: in-memory (slowapi default) for slowapi's leaky-bucket;
        SQLite-backed RateLimitService is used separately for free-tier gates.
    """
    from __future__ import annotations
    import ipaddress
    import logging
    from slowapi import Limiter
    from slowapi.errors import RateLimitExceeded
    from slowapi.util import get_remote_address
    from starlette.requests import Request
    from starlette.responses import JSONResponse

    from app.core.config import get_settings

    logger = logging.getLogger(__name__)


    def _client_subnet_key(request: Request) -> str:
        """Return /24 (IPv4) or /64 (IPv6) of the client IP."""
        settings = get_settings()
        raw_ip: str = ""
        if settings.auth.TRUST_CF_HEADER:
            raw_ip = request.headers.get("cf-connecting-ip", "") or request.headers.get("x-forwarded-for", "").split(",")[0].strip()
        if not raw_ip:
            raw_ip = get_remote_address(request)
        try:
            addr = ipaddress.ip_address(raw_ip)
        except ValueError:
            return raw_ip or "unknown"
        if isinstance(addr, ipaddress.IPv4Address):
            net = ipaddress.ip_network(f"{addr}/24", strict=False)
            return str(net.network_address) + "/24"
        net = ipaddress.ip_network(f"{addr}/64", strict=False)
        return str(net.network_address) + "/64"


    limiter = Limiter(key_func=_client_subnet_key, default_limits=[])


    async def rate_limit_handler(request: Request, exc: RateLimitExceeded) -> JSONResponse:
        """Format 429 with Retry-After header (RATE-12)."""
        retry_after = int(exc.detail.split()[-2]) if "second" in exc.detail else 60
        return JSONResponse(
            status_code=429,
            content={"detail": "Too many requests. Please try again later."},
            headers={"Retry-After": str(retry_after)},
        )
    ```

    Notes:
    - slowapi's in-memory bucket is acceptable for ANTI-01/02 because it's per-process (single-worker dev/prod) and resets on restart — matches the threat model (bot-scale attacks need durable bucket; single-process burst = adequate).
    - The SQLite-backed RateLimitService from Phase 11 is used separately in Plan 08 for FreeTierGate (per-user transcribe quotas).
    - If multi-worker is required later, swap slowapi storage to redis/limits — out of scope for v1.2.

    **2a. Create `app/api/auth_routes.py`** — fastapi APIRouter with 3 endpoints:

    ```python
    """Auth routes — register, login, logout (Phase 13).

    Exposes mailto:hey@logingrupa.lv as the password-reset path per AUTH-07.

    Per CONTEXT §74:
      - register: 3/hr/IP/24; generic 422 on duplicate or disposable email
      - login: 10/hr/IP/24; generic 401 on either wrong email OR wrong password
      - logout: idempotent; clears cookies; 204
    """
    from __future__ import annotations
    import logging
    from fastapi import APIRouter, Depends, Request, Response, status
    from app.api.dependencies import get_auth_service, get_csrf_service
    from app.api.schemas.auth_schemas import LoginRequest, RegisterRequest, AuthResponse
    from app.core.config import get_settings
    from app.core.disposable_email import is_disposable
    from app.core.exceptions import InvalidCredentialsError, UserAlreadyExistsError, ValidationError
    from app.core.rate_limiter import limiter
    from app.services.auth import AuthService, CsrfService

    logger = logging.getLogger(__name__)

    auth_router = APIRouter(
        prefix="/auth",
        tags=["Authentication"],
        responses={
            429: {"description": "Too many requests — see Retry-After header"},
        },
    )

    PASSWORD_RESET_HINT = "Password reset is manual — please email hey@logingrupa.lv (AUTH-07)."

    SESSION_COOKIE = "session"
    CSRF_COOKIE = "csrf_token"


    def _set_auth_cookies(response: Response, *, session_token: str, csrf_token: str) -> None:
        settings = get_settings()
        max_age = settings.auth.JWT_TTL_DAYS * 24 * 3600
        domain = settings.auth.COOKIE_DOMAIN or None
        response.set_cookie(
            key=SESSION_COOKIE,
            value=session_token,
            max_age=max_age,
            httponly=True,
            secure=settings.auth.COOKIE_SECURE,
            samesite="lax",
            path="/",
            domain=domain,
        )
        response.set_cookie(
            key=CSRF_COOKIE,
            value=csrf_token,
            max_age=max_age,
            httponly=False,
            secure=settings.auth.COOKIE_SECURE,
            samesite="lax",
            path="/",
            domain=domain,
        )


    def _clear_auth_cookies(response: Response) -> None:
        response.delete_cookie(SESSION_COOKIE, path="/")
        response.delete_cookie(CSRF_COOKIE, path="/")


    @auth_router.post(
        "/register",
        status_code=status.HTTP_201_CREATED,
        response_model=AuthResponse,
        description=f"Register a new user. {PASSWORD_RESET_HINT}",
    )
    @limiter.limit("3/hour")
    async def register(
        request: Request,
        response: Response,
        body: RegisterRequest,
        auth_service: AuthService = Depends(get_auth_service),
        csrf_service: CsrfService = Depends(get_csrf_service),
    ) -> AuthResponse:
        # ANTI-04: disposable email blocklist check (generic error)
        if is_disposable(body.email):
            logger.info("Registration rejected disposable_domain")
            raise ValidationError(message="Registration failed", code="REGISTRATION_FAILED", user_message="Registration failed")
        try:
            user = auth_service.register(body.email, body.password)
        except UserAlreadyExistsError as exc:
            # Anti-enumeration: same generic message as disposable rejection
            logger.info("Registration rejected duplicate id_unknown")
            raise ValidationError(message="Registration failed", code="REGISTRATION_FAILED", user_message="Registration failed") from exc
        # Issue session + csrf cookies (auto-login on register)
        session_token = auth_service.token_service.issue(int(user.id), user.token_version)
        csrf_token = csrf_service.issue()
        _set_auth_cookies(response, session_token=session_token, csrf_token=csrf_token)
        return AuthResponse(user_id=int(user.id), plan_tier=user.plan_tier)


    @auth_router.post(
        "/login",
        status_code=status.HTTP_200_OK,
        response_model=AuthResponse,
        description=f"Log in. {PASSWORD_RESET_HINT}",
    )
    @limiter.limit("10/hour")
    async def login(
        request: Request,
        response: Response,
        body: LoginRequest,
        auth_service: AuthService = Depends(get_auth_service),
        csrf_service: CsrfService = Depends(get_csrf_service),
    ) -> AuthResponse:
        try:
            user, session_token = auth_service.login(body.email, body.password)
        except InvalidCredentialsError:
            logger.info("Login rejected invalid_credentials")
            raise
        csrf_token = csrf_service.issue()
        _set_auth_cookies(response, session_token=session_token, csrf_token=csrf_token)
        return AuthResponse(user_id=int(user.id), plan_tier=user.plan_tier)


    @auth_router.post("/logout", status_code=status.HTTP_204_NO_CONTENT)
    async def logout(request: Request, response: Response) -> Response:
        # Idempotent — no-op if no session
        _clear_auth_cookies(response)
        return Response(status_code=status.HTTP_204_NO_CONTENT)
    ```

    Constraints:
    - SRP — routes do HTTP only; AuthService.register / .login do business logic
    - DRY — _set_auth_cookies + _clear_auth_cookies extracted helpers
    - No nested-if; flat try/except + early raise
    - Anti-enumeration: identical 422 message + code on disposable + duplicate
    - Logger NEVER logs email/password — only event labels
