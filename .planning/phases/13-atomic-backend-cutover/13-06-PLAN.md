    ```python
    """WS ticket service — 60-second single-use tickets for WebSocket auth (MID-06).

    Per CONTEXT §93: in-memory dict with TTL eviction; single-worker scope acceptable
    for v1.2 (single uvicorn worker). Multi-worker would require Redis or DB-backed
    storage — out of scope.

    Tickets are SHA-256-strong random 32-char urlsafe tokens. Cross-user check is
    enforced at consume-time (ticket carries user_id; caller passes task_id).
    """
    from __future__ import annotations
    import logging
    import secrets
    import threading
    from dataclasses import dataclass
    from datetime import datetime, timedelta, timezone

    logger = logging.getLogger(__name__)

    TICKET_TTL_SECONDS = 60
    TICKET_LENGTH = 32


    @dataclass
    class _Ticket:
        user_id: int
        task_id: str
        expires_at: datetime
        consumed: bool = False


    class WsTicketService:
        """In-memory single-use TTL ticket store. Thread-safe via Lock."""

        def __init__(self) -> None:
            self._tickets: dict[str, _Ticket] = {}
            self._lock = threading.Lock()

        def issue(self, *, user_id: int, task_id: str) -> tuple[str, datetime]:
            """Issue a fresh ticket valid for TICKET_TTL_SECONDS.

            Returns (token, expires_at). Token is urlsafe 32 chars (~190 bits).
            """
            self.cleanup_expired()
            token = secrets.token_urlsafe(24)[:TICKET_LENGTH]
            expires_at = datetime.now(timezone.utc) + timedelta(seconds=TICKET_TTL_SECONDS)
            with self._lock:
                self._tickets[token] = _Ticket(user_id=user_id, task_id=task_id, expires_at=expires_at)
            logger.debug("WsTicket issued user_id=%s task_id=%s", user_id, task_id)
            return token, expires_at

        def consume(self, token: str, task_id: str) -> int | None:
            """Atomically consume a ticket. Returns user_id on success, None on any failure.

            Failure modes (all return None — caller closes WS with 1008):
              - token not found
              - ticket expired
              - ticket already consumed (single-use)
              - ticket task_id != provided task_id (cross-user / cross-task)
            """
            now = datetime.now(timezone.utc)
            with self._lock:
                ticket = self._tickets.get(token)
                if ticket is None:
                    return None
                if ticket.consumed:
                    return None
                if ticket.expires_at < now:
                    return None
                if ticket.task_id != task_id:
                    return None
                ticket.consumed = True
                user_id = ticket.user_id
            logger.debug("WsTicket consumed user_id=%s task_id=%s", user_id, task_id)
            return user_id

        def cleanup_expired(self) -> int:
            """Best-effort eviction of expired tickets. Returns count removed."""
            now = datetime.now(timezone.utc)
            with self._lock:
                expired = [k for k, t in self._tickets.items() if t.expires_at < now or t.consumed]
                for key in expired:
                    del self._tickets[key]
            return len(expired)
    ```

    Constraints:
    - SRP — service does ticket lifecycle only
    - DRT — failure modes share the early-return pattern
    - No nested-if; flat guards inside the lock
    - Thread-safe — `with self._lock` guards both reads and writes (asyncio FastAPI runs single-threaded per worker, but threading.Lock is cheap insurance against future BackgroundTasks usage)

    ```python
    """POST /api/ws/ticket — issues a 60-second single-use WS ticket (MID-06).

    Caller must own the task_id (cross-user attempts return 404 opaque).
    """
    from __future__ import annotations
    import logging
    from fastapi import APIRouter, Depends, HTTPException, status

    from app.api.dependencies import get_authenticated_user, get_task_repository, _container, CONTAINER_NOT_INITIALIZED_ERROR
    from app.api.schemas.ws_ticket_schemas import TicketRequest, TicketResponse
    from app.domain.entities.user import User
    from app.domain.repositories.task_repository import ITaskRepository
    from app.services.ws_ticket_service import WsTicketService

    logger = logging.getLogger(__name__)

    ws_ticket_router = APIRouter(prefix="/api/ws", tags=["WebSocket"])


    def get_ws_ticket_service() -> WsTicketService:
        if _container is None:
            raise RuntimeError(CONTAINER_NOT_INITIALIZED_ERROR)
        return _container.ws_ticket_service()


    @ws_ticket_router.post("/ticket", response_model=TicketResponse, status_code=status.HTTP_201_CREATED)
    async def issue_ticket(
        body: TicketRequest,
        user: User = Depends(get_authenticated_user),
        repository: ITaskRepository = Depends(get_task_repository),
        ticket_service: WsTicketService = Depends(get_ws_ticket_service),
    ) -> TicketResponse:
        """Issue a WS ticket. Cross-user task_id → 404 opaque."""
        task = repository.get_by_id(body.task_id)
        if task is None:
            raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Task not found")
        if task.user_id != user.id:
            # Same opaque message — no enumeration of foreign tasks
            raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="Task not found")
        token, expires_at = ticket_service.issue(user_id=int(user.id), task_id=body.task_id)
        return TicketResponse(ticket=token, expires_at=expires_at)
    ```

    ```python
    """WebSocket API endpoint with single-use 60s ticket auth (MID-06/MID-07).

    Per CONTEXT §94: the WS endpoint accepts ?ticket=<token> query param,
    consumes the ticket atomically, and rejects with code=1008 on any failure
    (missing / expired / reused / cross-user).
    """
    import asyncio
    import logging
    from datetime import datetime, timezone

    from fastapi import APIRouter, Query, WebSocket, WebSocketDisconnect

    from app.core.container import Container
    from app.infrastructure.websocket import connection_manager
    from app.schemas.websocket_schemas import HeartbeatMessage

    logger = logging.getLogger(__name__)

    websocket_router = APIRouter(tags=["WebSocket"])

    HEARTBEAT_INTERVAL_SECONDS = 30
    WS_POLICY_VIOLATION = 1008


    @websocket_router.websocket("/ws/tasks/{task_id}")
    async def websocket_task_progress(
        websocket: WebSocket,
        task_id: str,
        ticket: str = Query(default=""),
    ) -> None:
        """WS endpoint with ticket validation. Closes 1008 on auth failure."""
        # Resolve container — websocket scope doesn't run dependency_injector wiring,
        # so we reach into the module-level container from app.api.dependencies.
        from app.api.dependencies import _container
        if _container is None:
            await websocket.close(code=WS_POLICY_VIOLATION)
            return

        ticket_service = _container.ws_ticket_service()
        if not ticket:
            await websocket.close(code=WS_POLICY_VIOLATION)
            return

        # Verify task exists + load its user_id
        task_repo = _container.task_repository()
        task = task_repo.get_by_id(task_id)
        if task is None:
            await websocket.close(code=WS_POLICY_VIOLATION)
            return

        # Atomically consume ticket; cross-user / expired / reused → None
        consumed_user_id = ticket_service.consume(ticket, task_id)
        if consumed_user_id is None:
            await websocket.close(code=WS_POLICY_VIOLATION)
            return
        if consumed_user_id != task.user_id:
            # Defense-in-depth — should not happen because consume() already checks task_id
            await websocket.close(code=WS_POLICY_VIOLATION)
            return

        # ---- ticket valid; proceed with the legacy connection-manager flow ----
        await connection_manager.connect(task_id, websocket)
        logger.info("WebSocket connection established for task %s user_id=%s", task_id, consumed_user_id)

        async def send_heartbeat_loop() -> None:
            while True:
                try:
                    await asyncio.sleep(HEARTBEAT_INTERVAL_SECONDS)
                    heartbeat = HeartbeatMessage(timestamp=datetime.now(timezone.utc))
                    await websocket.send_json(heartbeat.model_dump(mode="json"))
                except Exception as error:
                    logger.debug("Heartbeat loop ending for task %s: %s", task_id, str(error))
                    break

        heartbeat_task = asyncio.create_task(send_heartbeat_loop())

        try:
            while True:
                data = await websocket.receive_json()
                if data.get("type") == "ping":
                    await websocket.send_json({"type": "pong"})
        except WebSocketDisconnect:
            logger.info("WebSocket disconnected for task %s", task_id)
        except Exception as error:
            logger.warning("WebSocket error for task %s: %s", task_id, str(error))
        finally:
            heartbeat_task.cancel()
            try:
                await heartbeat_task
            except asyncio.CancelledError:
                pass
            await connection_manager.disconnect(task_id, websocket)
    ```

    Constraints:
    - 4 flat guards before connection accept (no nested-if)
    - Single 1008 close path — DRT
    - Reuse legacy heartbeat + ping/pong logic verbatim
    - Logger never logs ticket value

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-13-24 | Spoofing | Cross-user task_id at issue-time | mitigate | task.user_id == user.id check in POST /api/ws/ticket; opaque 404 on mismatch |
| T-13-25 | Replay | Reused ticket | mitigate | _Ticket.consumed flag set under lock; second consume() returns None → 1008 |

| T-13-29 | Information Disclosure | Ticket value in logs | mitigate | logger.debug logs user_id + task_id only; never the token value |
