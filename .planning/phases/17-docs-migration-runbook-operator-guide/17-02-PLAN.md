```
# === Auth (v1.2) ===

# v1.2 multi-user auth, API keys, and free-tier rate limits.

# Production-safety boot guard refuses to start with dev defaults — see app/core/feature_flags.py.

# ---- Auth secrets ----
# JWT signing secret for cookie session tokens (HS256). Generate with: openssl rand -hex 32
# Refuses boot in production with the placeholder default. Single decode site: app/core/jwt_codec.py.
JWT_SECRET=<change-me-in-production>

# CSRF double-submit cookie secret. Generate with: openssl rand -hex 32
# Refuses boot in production with the placeholder default. Single issuer: app/core/csrf_service.py.

# ---- Cookie & CORS ----
# Set to true in production behind HTTPS; false for local dev over http://localhost.
# AuthSettings validator refuses production boot when COOKIE_SECURE=false.
COOKIE_SECURE=false

# Cookie scope. Leave blank for localhost; set to your apex domain in production (e.g., .example.com).
COOKIE_DOMAIN=

# Browser origin allowed by CORS (allow_origins=[FRONTEND_URL] with allow_credentials=true).
# Must be a single explicit origin — never `*` while cookies are issued (see ANTI-06).
FRONTEND_URL=http://localhost:5173

# When true, slowapi key_func resolves CF-Connecting-IP / X-Forwarded-For for rate limiting.
# Set to true ONLY when the app sits behind Cloudflare or a trusted reverse proxy.
TRUST_CF_HEADER=false

# ---- Rate limit ----
# Per-IP-/24 cap on POST /auth/register attempts (ANTI-01). Default: 3.
RATE_LIMIT_REGISTER_PER_HOUR=3

# Per-IP-/24 cap on POST /auth/login attempts (ANTI-02). Default: 10.
RATE_LIMIT_LOGIN_PER_HOUR=10

# Per-user free-tier transcribe cap per hour (RATE-03). Default: 5.
RATE_LIMIT_FREE_REQ_PER_HOUR=5

# ---- Argon2 ----
# OWASP Argon2id parameters (AUTH-02). Tune up if Argon2 benchmark p99 < 100ms on your hardware.
# CI gate (VERIFY-05) asserts hash time stays under 300ms p99.
ARGON2_TIME_COST=2
ARGON2_MEMORY_KIB=19456
ARGON2_PARALLELISM=1

# ---- hCaptcha ----
# Feature flag — when false, the captcha hook is scaffolded but skipped (ANTI-05). Default: false.
HCAPTCHA_ENABLED=false

# Public site key from your hCaptcha dashboard. Leave blank while HCAPTCHA_ENABLED=false.
HCAPTCHA_SITE_KEY=

# Server-side secret. Leave blank while HCAPTCHA_ENABLED=false; required when ENABLED=true.
# Generate / retrieve from: https://dashboard.hcaptcha.com/sites
HCAPTCHA_SECRET=
```
